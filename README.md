## Install
```shell
npm install market-ui -S
```

## Quick Start
``` javascript
import Vue from 'vue'
import MarketUi from 'market-ui'

Vue.use(MarketUi)

// or
import {
  Button,
  Input
  // ...
} from 'market-ui'

Vue.use(Button, Input)
```
